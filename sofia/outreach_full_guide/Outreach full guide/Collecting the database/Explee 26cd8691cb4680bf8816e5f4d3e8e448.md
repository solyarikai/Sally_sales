**How to run**

1. Run the setup code first. 
    - Code
        
        ```jsx
        // Автоматический парсер Explee с защитой от Rate Limiting
        // Версия 2.4 - ИСПРАВЛЕН перехват (URL может быть объектом!)
        class ExpleeAutoParser {
          constructor() {
            this.lastPayload = null;
            this.isCapturing = false;
            this.originalFetch = window.fetch;
            this.progressKey = 'explee_parser_progress';
            this.startCapturing();
          }
        
          startCapturing() {
            if (this.isCapturing) return;
            
            const self = this;
            
            window.fetch = function(...args) {
              const [url, options] = args;
              
              // URL может быть строкой или объектом URL!
              if (options && options.method === 'POST' && options.body) {
                // Преобразуем URL в строку
                const urlString = typeof url === 'string' ? url : (url.href || url.toString());
                
                // Проверяем что это API Explee
                const isSearchAPI = 
                  urlString.includes('explee.com') && 
                  !urlString.includes('/track') && 
                  !urlString.includes('snitcher') &&
                  !urlString.includes('plausible');
                
                if (isSearchAPI) {
                  try {
                    const payload = JSON.parse(options.body);
                    
                    // Проверяем что это поисковый запрос (новая структура!)
                    if (payload.query && payload.filters) {
                      self.lastPayload = payload;
                      console.log('🎯 Payload перехвачен!');
                      console.log('   Query:', payload.query);
                      console.log('   Page:', payload.page || 1);
                      console.log('   Filters:', Object.keys(payload.filters).join(', '));
                    }
                  } catch (e) {
                    // Игнорируем ошибки парсинга
                  }
                }
              }
              
              return self.originalFetch.apply(this, args);
            };
            
            this.isCapturing = true;
            console.log('👁️ Начал перехват payload. Сделайте поиск на сайте, затем запустите парсер.');
          }
        
          stopCapturing() {
            window.fetch = this.originalFetch;
            this.isCapturing = false;
            console.log('🛑 Перехват остановлен');
          }
        
          getLastPayload() {
            if (this.lastPayload) {
              console.log(`✅ Используем перехваченный payload: "${this.lastPayload.query}"`);
              return this.lastPayload;
            }
            console.log('❌ Payload не найден!');
            console.log('');
            console.log('💡 Что попробовать:');
            console.log('   1. Сделайте поиск на сайте (введите запрос и нажмите Enter)');
            console.log('   2. Или измените любой фильтр');
            console.log('   3. Должно появиться: 🎯 Payload перехвачен!');
            return null;
          }
        
          // Сохранение прогресса
          saveProgress(data) {
            try {
              localStorage.setItem(this.progressKey, JSON.stringify(data));
            } catch (e) {
              console.warn('⚠️ Не удалось сохранить прогресс в localStorage');
            }
          }
        
          // Загрузка прогресса
          loadProgress() {
            try {
              const data = localStorage.getItem(this.progressKey);
              return data ? JSON.parse(data) : null;
            } catch (e) {
              return null;
            }
          }
        
          // Очистка прогресса
          clearProgress() {
            try {
              localStorage.removeItem(this.progressKey);
              console.log('🗑️ Прогресс очищен');
            } catch (e) {}
          }
        
          // Показать сохраненный прогресс
          showProgress() {
            const progress = this.loadProgress();
            if (!progress) {
              console.log('📭 Нет сохраненного прогресса');
              return;
            }
            
            console.log('📊 Сохраненный прогресс:');
            console.log(`   Запрос: "${progress.query}"`);
            console.log(`   Диапазон: ${progress.minSize}-${progress.maxSize} сотрудников`);
            console.log(`   Последний успешный диапазон: ${progress.lastSuccessfulRange}`);
            console.log(`   Всего компаний собрано: ${progress.totalCompanies}`);
            console.log(`   Дата: ${new Date(progress.timestamp).toLocaleString()}`);
            console.log('');
            console.log('💡 Для продолжения используйте:');
            console.log(`   parser.continueFromProgress()`);
            console.log('   или');
            console.log(`   parser.parseAuto(${progress.nextMinSize}, ${progress.maxSize})`);
          }
        
          // Функция для фильтрации email доменов (исключение личных почт)
          filterEmailDomains(emailText) {
            if (!emailText || emailText === '') return '';
            
            const emails = emailText.split(';').map(e => e.trim()).filter(e => e);
            
            const exactPersonalDomains = [
              'mail.ru', 'yandex.ru', 'yandex.com', 'rambler.ru', 'bk.ru', 'list.ru', 'inbox.ru',
              'icloud.com', 'me.com', 'mac.com',
              'live.com', 'live.ru', 'live.co.uk', 'msn.com',
              'yahoo.co.uk', 'yahoo.fr', 'yahoo.de', 'yahoo.es', 'yahoo.it', 'yahoo.ca', 'yahoo.com.au',
              'aol.com', 'wordpress.com', 'protonmail.com', 'tutanota.com', 'zoho.com', 'gmx.com', 
              'gmx.de', 'web.de', 'fastmail.com', 'mail.com', 'rediffmail.com', 'sina.com', 
              'qq.com', '163.com', '126.com', 'naver.com', 'daum.net', 'hanmail.net',
              't-online.de', 'freenet.de', '1und1.de',
              'laposte.net', 'orange.fr', 'free.fr', 'sfr.fr',
              'libero.it', 'virgilio.it', 'alice.it',
              'terra.es', 'movistar.es',
              'op.pl', 'o2.pl', 'interia.pl', 'seznam.cz', 'centrum.cz', 'email.cz',
              'terra.com', 'terra.com.br', 'uol.com.br', 'globo.com', 'ig.com.br',
              'nate.com', 'hanmail.net', 'yahoo.co.jp', 'goo.ne.jp'
            ];
            
            const personalDomainPrefixes = [
              'gmail', 'googlemail', 'outlook', 'hotmail', 'yahoo', 'ymail', 'rocketmail',
              'gmx', 'mail'
            ];
            
            for (let email of emails) {
              if (email.includes('@')) {
                const domain = email.split('@')[1].trim().toLowerCase();
                
                if (domain.length === 0) continue;
                if (exactPersonalDomains.includes(domain)) continue;
                
                const startsWithPersonal = personalDomainPrefixes.some(prefix => domain.startsWith(prefix));
                if (startsWithPersonal) continue;
                
                if (domain.includes('mail') && domain.length < 15) continue;
                if (domain.includes('email') && domain.length < 20) continue;
                
                return domain;
              }
            }
            
            return '';
          }
        
          convertLinkedInId(linkedinId) {
            if (!linkedinId || linkedinId === '' || linkedinId === null) return '';
            return `http://www.linkedin.com/company/${linkedinId}`;
          }
        
          processCompany(company) {
            const processed = { ...company };
            
            // Заполнение domain из email
            if (!processed.domain || processed.domain === '') {
              if (processed.Emails && Array.isArray(processed.Emails)) {
                const emailsString = processed.Emails.join('; ');
                const extractedDomain = this.filterEmailDomains(emailsString);
                if (extractedDomain) {
                  processed.domain = extractedDomain;
                }
              }
            }
            
            // LinkedIn URL
            if (processed.linkedin_id) {
              processed.linkedin_url = this.convertLinkedInId(processed.linkedin_id);
            } else {
              processed.linkedin_url = '';
            }
            
            // Преобразование массивов в строки
            if (Array.isArray(processed.Emails)) {
              processed.Emails = processed.Emails.join('; ');
            }
            
            if (Array.isArray(processed['Founders Origin'])) {
              processed['Founders Origin'] = processed['Founders Origin'].join('; ');
            }
            
            // Employees by Country в отдельные колонки
            if (processed['Employees by Country'] && typeof processed['Employees by Country'] === 'object') {
              const employeesByCountry = processed['Employees by Country'];
              for (let country in employeesByCountry) {
                processed[`Employees_${country}`] = employeesByCountry[country];
              }
              delete processed['Employees by Country'];
            }
            
            return processed;
          }
        
          convertToCSV(companies) {
            if (!companies || companies.length === 0) return '';
            
            const processedCompanies = companies.map(c => this.processCompany(c));
            
            const allKeys = new Set();
            processedCompanies.forEach(company => {
              Object.keys(company).forEach(key => allKeys.add(key));
            });
            
            const headers = Array.from(allKeys);
            const csvHeaders = headers.map(h => this.escapeCSV(h)).join(',');
            
            const csvRows = processedCompanies.map(company => {
              return headers.map(header => {
                const value = company[header];
                if (value === null || value === undefined) return '';
                return this.escapeCSV(String(value));
              }).join(',');
            });
            
            return [csvHeaders, ...csvRows].join('\n');
          }
        
          escapeCSV(value) {
            if (value === null || value === undefined) return '';
            const str = String(value);
            if (str.includes(',') || str.includes('"') || str.includes('\n')) {
              return '"' + str.replace(/"/g, '""') + '"';
            }
            return str;
          }
        
          // Функция для генерации случайной задержки (jitter)
          getRandomDelay(baseDelay, variance = 0.3) {
            const min = baseDelay * (1 - variance);
            const max = baseDelay * (1 + variance);
            return Math.floor(Math.random() * (max - min + 1)) + min;
          }
        
          // Продолжить с сохраненного прогресса
          async continueFromProgress() {
            const progress = this.loadProgress();
            if (!progress) {
              console.log('❌ Нет сохраненного прогресса для продолжения');
              console.log('💡 Запустите новый парсинг: parser.parseAuto(1, 500)');
              return null;
            }
        
            console.log('♻️ Продолжаю парсинг с сохраненного прогресса...');
            console.log(`📍 Начинаю с диапазона: ${progress.nextMinSize}-${progress.maxSize}`);
            
            const existingCompanies = progress.companies || [];
            console.log(`📦 Загружено ${existingCompanies.length} компаний из предыдущей сессии`);
        
            return await this.parseWithPayload(
              progress.payload,
              progress.nextMinSize,
              progress.maxSize,
              progress.maxPagesPerRange,
              existingCompanies,
              progress.allStats || []
            );
          }
        
          async parseAuto(minSize = 1, maxSize = 10000, maxPagesPerRange = 100, delayBetweenPages = 2000, delayBetweenRanges = 3000) {
            const basePayload = this.getLastPayload();
            
            if (!basePayload) {
              return null;
            }
        
            return await this.parseWithPayload(basePayload, minSize, maxSize, maxPagesPerRange, [], [], delayBetweenPages, delayBetweenRanges);
          }
        
          async parseWithPayload(basePayload, minSize, maxSize, maxPagesPerRange, existingCompanies = [], existingStats = [], delayBetweenPages = 2000, delayBetweenRanges = 3000) {
            const baseUrl = "https://api.explee.com/api/search/companies";
            
            const headers = {
              "accept": "*/*",
              "accept-encoding": "gzip, deflate, br, zstd",
              "accept-language": "ru,en-US;q=0.9,en;q=0.8",
              "content-type": "application/json",
              "origin": "https://explee.com",
              "referer": "https://explee.com/",
              "sec-fetch-dest": "empty",
              "sec-fetch-mode": "cors",
              "sec-fetch-site": "same-site",
              "user-agent": navigator.userAgent,
              "x-device-id": "1aa3f4a4-1d04-4c70-a709-db508f9d6cd4"
            };
        
            function generateSizeRanges(min, max) {
              const ranges = [];
              let current = min;
              
              while (current <= max) {
                let step;
                if (current <= 10) step = 2;
                else if (current <= 50) step = 5;
                else if (current <= 200) step = 25;
                else if (current <= 1000) step = 100;
                else if (current <= 5000) step = 500;
                else step = 1000;
                
                const rangeMax = Math.min(current + step - 1, max);
                ranges.push({
                  min: current,
                  max: rangeMax,
                  name: `${current}-${rangeMax}`
                });
                
                current = rangeMax + 1;
              }
              
              return ranges;
            }
        
            const sizeRanges = generateSizeRanges(minSize, maxSize);
            let allCompanies = [...existingCompanies];
            let allStats = [...existingStats];
        
            console.log(`🚀 Автопарсинг запроса: "${basePayload.query}"`);
            console.log(`📊 Размер компаний: ${minSize}-${maxSize} сотрудников`);
            console.log(`🔧 Диапазонов: ${sizeRanges.length}, макс страниц на диапазон: ${maxPagesPerRange}`);
            console.log(`⏱️ Задержки: ${delayBetweenPages}мс между страницами, ${delayBetweenRanges}мс между диапазонами`);
        
            const self = this;
        
            // Функция с retry при 429
            async function createSearchWithRetry(payload, retryCount = 0, maxRetries = 3) {
              try {
                console.log('  🔄 Отправляю POST запрос...');
                const response = await self.originalFetch(baseUrl, {
                  method: "POST",
                  mode: "cors",
                  credentials: "include",
                  headers: headers,
                  body: JSON.stringify(payload)
                });
        
                console.log('  📡 POST response status:', response.status);
        
                if (response.status === 429) {
                  if (retryCount < maxRetries) {
                    const waitTime = Math.min(30000 * Math.pow(2, retryCount), 120000);
                    console.log(`  ⏰ Rate limit (429)! Попытка ${retryCount + 1}/${maxRetries}. Жду ${waitTime/1000} секунд...`);
                    await new Promise(resolve => setTimeout(resolve, waitTime));
                    return await createSearchWithRetry(payload, retryCount + 1, maxRetries);
                  } else {
                    console.log('  ❌ Превышено количество попыток retry. Rate limit все еще активен.');
                    return { error: 'rate_limit', status: 429 };
                  }
                }
        
                if (!response.ok) {
                  throw new Error(`HTTP error! status: ${response.status}`);
                }
        
                const data = await response.json();
                
                if (data.token) {
                  console.log('  ✅ Найден token:', data.token);
                  return data.token;
                }
                
                if (data.searchId || data.id || data.hash) {
                  const hash = data.searchId || data.id || data.hash;
                  console.log('  ✅ Найден хэш:', hash);
                  return hash;
                }
                
                const responseUrl = response.url;
                const hashMatch = responseUrl.match(/\/([a-f0-9]{32})$/);
                if (hashMatch) {
                  console.log('  ✅ Хэш извлечен из URL:', hashMatch[1]);
                  return hashMatch[1];
                }
                
                if (data.companies) {
                  console.log('  ✅ POST вернул данные напрямую:', data.companies.length, 'компаний');
                  return data;
                }
                
                console.log('  ⚠️ Хэш не найден, структура данных:', Object.keys(data));
                return data;
                
              } catch (error) {
                console.error('  ❌ Ошибка создания поиска:', error);
                return null;
              }
            }
        
            async function fetchByToken(token, maxRetries = 10) {
              const url = `${baseUrl}/${token}`;
              
              for (let attempt = 1; attempt <= maxRetries; attempt++) {
                try {
                  console.log(`  🔍 GET запрос (попытка ${attempt}):`, url);
                  const response = await self.originalFetch(url, {
                    method: "GET",
                    mode: "cors",
                    credentials: "include",
                    headers: headers
                  });
        
                  if (!response.ok) {
                    console.log('  ❌ GET error:', response.status);
                    return null;
                  }
        
                  const data = await response.json();
                  
                  if (data.status === 'pending') {
                    console.log(`  ⏳ Поиск обрабатывается, ждем 2 сек...`);
                    await new Promise(resolve => setTimeout(resolve, 2000));
                    continue;
                  }
                  
                  console.log('  📦 GET data:', data.companies ? `${data.companies.length} компаний` : 'нет companies');
                  return data;
                } catch (error) {
                  console.error('  ❌ Ошибка получения данных:', error);
                  return null;
                }
              }
              
              console.log('  ⏰ Превышено время ожидания (поиск все еще pending)');
              return null;
            }
        
            async function parseRange(range) {
              console.log(`\n📈 Диапазон ${range.name} сотрудников:`);
              
              const rangePayload = JSON.parse(JSON.stringify(basePayload));
              
              // Новая структура! filters.size вместо старой
              if (!rangePayload.filters) {
                rangePayload.filters = {};
              }
              rangePayload.filters.size = {
                "min": range.min,
                "max": range.max
              };
        
              let rangeCompanies = [];
              let page = 1;
              
              while (page <= maxPagesPerRange) {
                rangePayload.page = page;
                
                const searchResult = await createSearchWithRetry(rangePayload);
                
                if (searchResult && searchResult.error === 'rate_limit') {
                  return { error: 'rate_limit', range: range };
                }
                
                if (!searchResult) {
                  console.log(`  ❌ Не удалось создать поиск для страницы ${page}`);
                  break;
                }
        
                const token = typeof searchResult === 'string' ? searchResult : searchResult.token;
                
                if (!token) {
                  console.log(`  ❌ Не найден token для страницы ${page}`);
                  break;
                }
        
                const data = await fetchByToken(token);
                
                if (!data || !data.companies || data.companies.length === 0) {
                  console.log(`  ⏹️ Страница ${page}: нет данных`);
                  break;
                }
                
                rangeCompanies.push(...data.companies);
                console.log(`  📄 Страница ${page}: +${data.companies.length} компаний (в диапазоне: ${rangeCompanies.length})`);
                
                if (data.companies.length < 10) break;
                page++;
                
                const delay = self.getRandomDelay(delayBetweenPages);
                await new Promise(resolve => setTimeout(resolve, delay));
              }
        
              return {
                range: range.name,
                companies: rangeCompanies,
                pages: page - 1,
                count: rangeCompanies.length
              };
            }
        
            // Парсим все диапазоны
            for (let i = 0; i < sizeRanges.length; i++) {
              const range = sizeRanges[i];
              const rangeResult = await parseRange(range);
              
              if (rangeResult.error === 'rate_limit') {
                console.log('\n🚨 ОСТАНОВКА: Достигнут rate limit!');
                console.log('━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━');
                console.log(`📍 Последний успешный диапазон: ${i > 0 ? sizeRanges[i-1].name : 'нет'}`);
                console.log(`⏸️ Остановился на диапазоне: ${range.name}`);
                console.log(`📊 Всего собрано: ${allCompanies.length} компаний`);
                console.log('');
                console.log('💡 ЧТО ДЕЛАТЬ:');
                console.log('   1. Подключите VPN или смените IP адрес');
                console.log('   2. Подождите 20-30 минут');
                console.log('   3. Запустите продолжение:');
                console.log('');
                console.log(`   parser.parseAuto(${range.min}, ${maxSize})`);
                console.log('   или');
                console.log(`   parser.continueFromProgress()`);
                console.log('━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━');
                
                self.saveProgress({
                  payload: basePayload,
                  minSize: minSize,
                  maxSize: maxSize,
                  maxPagesPerRange: maxPagesPerRange,
                  lastSuccessfulRange: i > 0 ? sizeRanges[i-1].name : 'нет',
                  nextMinSize: range.min,
                  companies: allCompanies,
                  totalCompanies: allCompanies.length,
                  allStats: allStats,
                  timestamp: new Date().toISOString(),
                  query: basePayload.query
                });
                
                if (allCompanies.length > 0) {
                  await self.savePartialResults(allCompanies, basePayload, minSize, maxSize, 'partial');
                }
                
                return null;
              }
              
              if (rangeResult.companies.length > 0) {
                allCompanies.push(...rangeResult.companies);
                allStats.push({
                  range: rangeResult.range,
                  companies_count: rangeResult.count,
                  pages_parsed: rangeResult.pages
                });
                
                console.log(`✅ Диапазон ${rangeResult.range}: ${rangeResult.count} компаний`);
                
                self.saveProgress({
                  payload: basePayload,
                  minSize: minSize,
                  maxSize: maxSize,
                  maxPagesPerRange: maxPagesPerRange,
                  lastSuccessfulRange: rangeResult.range,
                  nextMinSize: i < sizeRanges.length - 1 ? sizeRanges[i + 1].min : maxSize,
                  companies: allCompanies,
                  totalCompanies: allCompanies.length,
                  allStats: allStats,
                  timestamp: new Date().toISOString(),
                  query: basePayload.query
                });
              } else {
                console.log(`⚪ Диапазон ${rangeResult.range}: пустой`);
              }
              
              if (i < sizeRanges.length - 1) {
                const delay = self.getRandomDelay(delayBetweenRanges);
                console.log(`⏱️ Пауза ${delay}мс перед следующим диапазоном...`);
                await new Promise(resolve => setTimeout(resolve, delay));
              }
            }
        
            // Удаляем дубликаты
            const uniqueCompanies = [];
            const seenDomains = new Set();
            
            for (let company of allCompanies) {
              if (company.domain && !seenDomains.has(company.domain)) {
                uniqueCompanies.push(company);
                seenDomains.add(company.domain);
              } else if (!company.domain) {
                uniqueCompanies.push(company);
              }
            }
        
            const duplicatesRemoved = allCompanies.length - uniqueCompanies.length;
        
            console.log(`\n🎯 ИТОГИ:`);
            console.log(`📊 Всего собрано: ${allCompanies.length} компаний`);
            console.log(`🔄 Удалено дубликатов: ${duplicatesRemoved}`);
            console.log(`✨ Уникальных компаний: ${uniqueCompanies.length}`);
            console.log(`\n🔧 Обработка данных для CSV...`);
        
            const csvContent = this.convertToCSV(uniqueCompanies);
        ```

    ## ✨ НОВЫЕ ВОЗМОЖНОСТИ
    
    ### 1. **CSV вместо JSON**

    ### 2. **Автозаполнение Domain из Email**
    
    **Было:** Пустые domain приходилось заполнять вручную
    
    **Стало:** Автоматическое извлечение корпоративных доменов из email
    
    **Логика:**
    
    - Исключает личные почты (gmail, outlook, yahoo, mail.ru и т.д.)
    - Берет первый корпоративный домен
    - Заполняет пустые ячейки domain
    
    **Пример:**
    
    `Email: john@gmail.com; mary@company.com
    → domain: company.com`
    
    **Поддерживаются 50+ личных доменов для фильтрации**
    
    ### 3. **Преобразование LinkedIn ID в URL**
    
    **Было:** Просто число в колонке
    
    **Стало:** Готовая кликабельная ссылка
    
    **Преобразование:**
    
    `Было: 82570895
    Стало: http://www.linkedin.com/company/82570895`
    
    Новая колонка: `linkedin_url`
    
    ### 4. **Защита от Rate Limiting (429)**
    
    ### 4.1 Автоматический Retry
    
    **Было:** Парсер падал при первом 429
    
    **Стало:** До 3 попыток с увеличивающимися паузами
    
    **Логика:**
    
    `429 → Пауза 30 сек → Retry
    429 → Пауза 60 сек → Retry
    429 → Пауза 120 сек → Остановка с сохранением прогресса`
    
    ### 4.2 Сохранение прогресса
    
    **Было:** При 429 терялся весь прогресс
    
    **Стало:** Автоматическое сохранение после каждого диапазона
    
    **В localStorage сохраняется:**
    
    - Все собранные компании
    - Последний успешный диапазон
    - Настройки парсинга (payload, параметры)
    - Статистика по диапазонам
    
    ### 4.3 Продолжение с любого места
    
    **Новая команда:**
    
    javascript
    
    `parser.continueFromProgress()
    *// Автоматически продолжает с сохраненного места*`
    
    **Также можно вручную:**
    
    javascript
    
    `parser.parseAuto(200, 500)  *// С диапазона где остановились*`
    
    ### 5. **Умные задержки**
    
    ### 5.1 Настраиваемые задержки
    
    **Было:** Фиксированные 1200мс между страницами
    
    **Стало:** Гибкие настройки
    
    **Параметры:**
    
    javascript

    )`
    
    ### 5.2 Jitter (случайные вариации)
    
    **Стало:** Задержки варьируются ±30%
    
    **Зачем:** Имитация человеческого поведения
    
    **Пример:**
    
    `Задержка 3000мс → реально 2100-3900мс`
    
    ### 6. **Частичные результаты**
    
    **Было:** При 429 терялось всё
